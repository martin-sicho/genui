[docs]    def create(self, validated_data):
        ret = super().create(validated_data)

        for app in apps.all_():
            runTask(
                    createDefaultModels,
                    eager=hasattr(settings, 'CELERY_TASK_ALWAYS_EAGER') and settings.CELERY_TASK_ALWAYS_EAGER,
                    args=(
                        ret.id,
                        app
                    ),
                )

        return ret
